class Solution {
  public:
    int totalCount(int k, vector<int>& arr) {
        int ans = 0;
        for(int i = 0; i < arr.size(); i++){
            if(arr[i] % k == 0) {
                ans += (arr[i] / k);
            } else {
                ans += (arr[i] / k) + 1;
            }
        }
        return ans;
    }
};
